## 【代码实现】

```java
import java.util.Stack;
public class StackTest2 {

    public static void main(String[] args) {
        Stack<Integer> stack = new Stack<>();
        stack.push(2);
        stack.push(1);
        stack.push(5);
        stack.push(4);
        stack.push(3);

        Stack<Integer> help = new Stack<>();

        sortForStack(stack, help);


    }

    private static void sortForStack(Stack<Integer> stack, Stack<Integer> help) {

        while (!stack.isEmpty()) {
            if (help.isEmpty()) {
                help.push(stack.pop());
            }
            int current = stack.pop();
            if (help.peek() <= current) {
                help.push(current);
            } else {
                while ((!help.isEmpty())&&help.peek() > current) {
                    stack.push(help.pop());
                }
                help.push(current);
            }
        }

        while (!help.isEmpty()) {
            stack.push(help.pop());
        }

        while (!stack.isEmpty()) {
            System.out.println(stack.pop());
        }


    }

}

```


运行结果：
```
1
2
3
4
5
```
